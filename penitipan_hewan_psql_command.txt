CREATE DATABASE penitipan_hewan;

CREATE TABLE pemilik(
NIK VARCHAR(16) NOT NULL,
nama VARCHAR(50) NOT NULL,
no_telepon INT NOT NULL,
alamat VARCHAR(101) NOT NULL,
email VARCHAR(20) NOT NULL,
id_hewan serial,
CONSTRAINT NIK_PK PRIMARY KEY(NIK);
CONSTRAINT hewan_id_FK FOREIGN KEY REFERENCES hewan(hewan_id));

CREATE TABLE hewan(
id_hewan serial,
jenis_hewan VARCHAR(50) NOT NULL,
umur INT NOT NULL,
penyakit VARCHAR(50),
CONSTRAINT hewan_id_PK PRIMARY KEY(id_hewan));

CREATE TABLE petugas(
id_petugas serial PRIMARY KEY,
nama VARCHAR(50) NOT NULL,
no_telepon INT NOT NULL,
alamat VARCHAR(50) NOT NULL,
email VARCHAR(50) NOT NULL);

CREATE TABLE data_penitipan(
tanggal_penerimaan DATE NOT NULL,
tanggal_pengembalian DATE NOT NULL,
biaya_penitipan INT NOT NULL,
nik_pemilik VARCHAR(16) NOT NULL,
id_petugas serial,
CONSTRAINT nik_pemilik_FK FOREIGN KEY(nik_pemilik) REFERENCES pemilik(nik),
CONSTRAINT id_petugas_FK FOREIGN KEY(id_petugas) REFERENCES petugas(id_petugas));
